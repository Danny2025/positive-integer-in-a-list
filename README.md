# positive-integer-in-a-list
python program to print the positive numbers from the list
myList1 = [ 12, -7, 5, 64, -14]
print('positive numbers:')
for number in myList1:
    if number >= 0:
       print(number) 

myList2 = [12, 14, -95, 3 ]
print('positive numbers:')
for num in myList2:
